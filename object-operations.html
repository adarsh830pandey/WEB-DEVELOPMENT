🚀 Object Operations in JavaScript

1. Creating Objects

// Object literal let user = { name: 'John', age: 30 }; // Using constructor let car = new Object(); car.brand = 'Toyota'; // Using Object.create() let obj = Object.create(null); // Using class class Person { constructor(name) { this.name = name; } }

2. Accessing Properties

let user = { name: 'John', age: 30, 'full-name': 'John Doe' }; user.name // dot notation user['age'] // bracket notation user['full-name'] // for properties with hyphens user.email || 'N/A' // default value if undefined

3. Adding and Modifying Properties

let obj = { name: 'John' }; obj.age = 30 // add property obj['city'] = 'NYC' // add property using bracket obj.name = 'Jane' // modify property delete obj.age // delete property obj.email = undefined // set to undefined

4. Object Methods

let obj = { name: 'John', age: 30 }; Object.keys(obj) // ['name', 'age'] Object.values(obj) // ['John', 30] Object.entries(obj) // [['name', 'John'], ['age', 30]] Object.assign(obj, {city: 'NY'}) // merge objects

5. Checking Properties

let user = { name: 'John', age: 30 }; 'name' in user // true user.hasOwnProperty('age') // true user.hasOwnProperty('city') // false typeof user.email // 'undefined'

6. Iterating Through Objects

let user = { name: 'John', age: 30, city: 'NYC' }; for (let key in user) { } // for...in loop Object.keys(user).forEach(key => {}) // forEach with keys Object.entries(user).forEach(([k, v]) => {}) // forEach entries

7. Nested Objects

let user = { name: 'John', address: { city: 'NYC', zip: '10001' }, hobbies: ['reading', 'gaming'] }; user.address.city // 'NYC' user.address['zip'] // '10001' user.hobbies[0] // 'reading'

8. Object Spreading & Cloning

let obj1 = { name: 'John', age: 30 }; let obj2 = { city: 'NYC' }; // Shallow copy let copy = { ...obj1 }; let copy2 = Object.assign({}, obj1); // Merge objects let merged = { ...obj1, ...obj2 }; // Override properties let updated = { ...obj1, age: 31 };

9. Object Destructuring

let user = { name: 'John', age: 30, city: 'NYC' }; // Basic destructuring let { name, age } = user; // Renaming let { name: userName, age: userAge } = user; // Default values let { email = 'no-email' } = user; // Rest properties let { name, ...rest } = user;

10. Methods in Objects

let user = { name: 'John', greet: function() { return 'Hello ' + this.name; }, introduce() { return 'I am ' + this.name; }, age: 30 }; user.greet() // 'Hello John' user.introduce() // 'I am John'

11. Getters and Setters

let user = { _age: 30, get age() { return this._age; }, set age(value) { if (value > 0) this._age = value; } }; user.age // getter user.age = 31 // setter

12. Freezing and Sealing Objects

let user = { name: 'John', age: 30 }; // Freeze - no changes allowed Object.freeze(user); // Seal - modify existing, no add/delete Object.seal(user); // Check if frozen/sealed Object.isFrozen(user) Object.isSealed(user)

13. Classes and Constructor Functions

class Person { constructor(name, age) { this.name = name; this.age = age; } greet() { return 'Hi, I am ' + this.name; } } let person1 = new Person('John', 30);

14. Defining Properties with Property Descriptors

let obj = {}; Object.defineProperty(obj, 'name', { value: 'John', writable: true, // can be modified enumerable: true, // shows in for...in configurable: true // can be deleted/redefined }); Object.getOwnPropertyDescriptor(obj, 'name')
