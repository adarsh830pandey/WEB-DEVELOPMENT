🚀 Array Operations in JavaScript

1. Creating Arrays

// Using array literal let fruits = ['Apple', 'Banana', 'Orange']; // Using Array constructor let numbers = new Array(1, 2, 3, 4, 5); // Using Array.from() let range = Array.from({length: 5}, (_, i) => i + 1);

2. Accessing Array Elements

let arr = ['a', 'b', 'c', 'd']; arr[0] // 'a' - first element arr[2] // 'c' - third element arr.at(-1) // 'd' - last element arr[arr.length - 1] // 'd' - last element

3. Adding and Removing Elements

let arr = [1, 2, 3]; arr.push(4, 5) // Add to end arr.pop() // Remove from end arr.unshift(0) // Add to beginning arr.shift() // Remove from beginning arr.splice(1, 2) // Remove 2 elements starting at index 1

4. Iterating Through Arrays

let arr = [1, 2, 3]; arr.forEach(item => console.log(item)) arr.map(x => x * 2) arr.filter(x => x > 1) for (let i = 0; i < arr.length; i++) { } for (let item of arr) { }

5. Transforming Arrays (map, filter, reduce)

let numbers = [1, 2, 3, 4, 5]; numbers.map(x => x * 2) // [2, 4, 6, 8, 10] numbers.filter(x => x > 2) // [3, 4, 5] numbers.reduce((sum, x) => sum + x, 0) // 15

6. Searching Arrays

let arr = ['a', 'b', 'c', 'b']; arr.indexOf('b') // 1 arr.includes('c') // true arr.find(x => x === 'b') // 'b' arr.findIndex(x => x === 'b') // 1

7. Combining and Splitting Arrays

let arr1 = [1, 2]; let arr2 = [3, 4]; arr1.concat(arr2) // [1, 2, 3, 4] [...arr1, ...arr2] // [1, 2, 3, 4] - spread operator arr1.join('-') // '1-2' 'a-b-c'.split('-') // ['a', 'b', 'c']

8. Sorting and Reversing

let arr = [3, 1, 4, 1, 5]; arr.sort() // [1, 1, 3, 4, 5] arr.sort((a, b) => b - a) // [5, 4, 3, 1, 1] - descending arr.reverse() // Reverses array

9. Checking Array Conditions

let arr = [2, 4, 6, 8]; arr.every(x => x % 2 === 0) // true - all even arr.some(x => x > 5) // true - some > 5 arr.length // 4 Array.isArray(arr) // true

10. Array Destructuring

let [a, b, c] = [1, 2, 3] let [first, ...rest] = [1, 2, 3, 4] // rest = [2, 3, 4] let [x, , z] = [1, 2, 3] // skips middle element
