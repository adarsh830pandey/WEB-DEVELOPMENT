🎪 Event Handling in JavaScript

1. Click Events

// Inline event handler <button onclick="handleClick()">Click Me</button> // addEventListener element.addEventListener('click', function() { console.log('Clicked!') }) // Single vs Double click element.addEventListener('click', handleClick) element.addEventListener('dblclick', handleDoubleClick)

2. Mouse Events

element.addEventListener('mouseover', handler) element.addEventListener('mouseout', handler) element.addEventListener('mousemove', handler) element.addEventListener('mousedown', handler) element.addEventListener('mouseup', handler) element.addEventListener('mouseenter', handler) element.addEventListener('mouseleave', handler)

3. Keyboard Events

element.addEventListener('keydown', handler) element.addEventListener('keyup', handler) element.addEventListener('keypress', handler) // Event properties event.key // Character pressed event.code // Key code event.keyCode // Numeric code (deprecated) event.ctrlKey // Ctrl pressed? event.shiftKey // Shift pressed?

4. Form Events

input.addEventListener('focus', handler) input.addEventListener('blur', handler) input.addEventListener('change', handler) input.addEventListener('input', handler) form.addEventListener('submit', handler)

5. Event Object Properties

element.addEventListener('click', function(event) { event.type // 'click' event.target // clicked element event.currentTarget // element listener attached to event.timeStamp // when event occurred event.clientX, clientY // mouse coordinates event.preventDefault() // stop default behavior })

6. preventDefault & stopPropagation

// Prevent default behavior event.preventDefault() // block link navigation, form submit, etc // Stop event bubbling event.stopPropagation() // stop event from bubbling up // Stop event and all handlers event.stopImmediatePropagation()
Link (preventDefault)

7. Event Bubbling & Capturing

// Bubbling (default) - events bubble up element.addEventListener('click', handler) // Capturing - events captured going down element.addEventListener('click', handler, true) // Event phases event.CAPTURING_PHASE // 1 event.AT_TARGET // 2 event.BUBBLING_PHASE // 3

8. Event Delegation

// Instead of adding listener to each item, // add one to parent and check event.target parent.addEventListener('click', function(event) { if (event.target.matches('.item')) { // Handle item click } })

9. Adding & Removing Event Listeners

// Add listener element.addEventListener('click', myHandler) // Remove listener (need same function reference) element.removeEventListener('click', myHandler) // One-time listener element.addEventListener('click', handler, { once: true })

10. Custom Events

// Create custom event let event = new CustomEvent('myEvent', { detail: { message: 'Hello' } }) // Dispatch event element.dispatchEvent(event) // Listen for custom event element.addEventListener('myEvent', function(e) { console.log(e.detail.message) })

11. Window Events

window.addEventListener('load', handler) window.addEventListener('beforeunload', handler) window.addEventListener('resize', handler) window.addEventListener('scroll', handler) window.addEventListener('orientationchange', handler)

12. Working with setTimeout/setInterval

// Run once after delay setTimeout(function() { console.log('Executed after 2 seconds') }, 2000) // Run repeatedly let id = setInterval(function() { console.log('Repeated every 1 second') }, 1000) // Stop interval clearInterval(id) clearTimeout(timerId)

13. Debouncing & Throttling

// Debounce - wait for pause before executing function debounce(func, delay) { let timerId; return function() { clearTimeout(timerId); timerId = setTimeout(func, delay); }; } // Use it input.addEventListener('input', debounce(handleInput, 500))
